def _launch():
    return ((1, 1, 1), (1, 1, 1))
INPUT0_SHAPE = (16, 64, 512, 512)
OUTPUT_SHAPE = (16, 128, 512, 512)

@substrate.jit
def fused_kernel(X: S.Tensor((16, 64, 512, 512), S.f32), DW: S.Tensor((64, 1, 3, 3), S.f32), PW: S.Tensor((128, 64, 1, 1), S.f32), Y: S.Tensor((16, 128, 512, 512), S.f32)):
    for n in S.range(16):
        for oc in S.range(128):
            for oh in S.range(512):
                for ow in S.range(512):
                    acc = S.convert(0.0, S.f32)
                    for ic in S.range(64):
                        tmp = S.convert(0.0, S.f32)
                        for kh in S.range(3):
                            for kw in S.range(3):
                                ih = oh * 1 - 1 + kh * 1
                                iw = ow * 1 - 1 + kw * 1
                                if ih >= 0 and ih < 512 and iw >= 0 and iw < 512:
                                    tmp += S.convert(X[n, ic, ih, iw], S.f32) * S.convert(DW[ic, 0, kh, kw], S.f32)
                        acc += tmp * S.convert(PW[oc, ic, 0, 0], S.f32)
                    Y[n, oc, oh, ow] = S.convert(acc, S.f32)

class ModelNew(nn.Module):

    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
            super(ModelNew, self).__init__()
            self.depthwise = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, dilation=dilation, groups=in_channels, bias=bias)
            self.pointwise = nn.Conv2d(in_channels, out_channels, kernel_size=1, bias=bias)

    def forward(self, x):
        if tuple(x.shape) != (16, 64, 512, 512) or x.dtype != torch.float32:
            raise RuntimeError('This fused kernel only supports the benchmark input shape and dtype.')
        x0 = x.contiguous()
        dw = self.depthwise.weight.to(device=x.device, dtype=x.dtype).contiguous()
        pw = self.pointwise.weight.to(device=x.device, dtype=x.dtype).contiguous()
        y = torch.empty((16, 128, 512, 512), device=x.device, dtype=x.dtype)
        fused_kernel[_launch](x0, dw, pw, y)
        return y

def _ensure_rocm_available(label):
    """Check that ROCm/HIP is available."""
    if not torch.cuda.is_available():
        raise RuntimeError("CUDA is not available.")
    if not hasattr(torch.version, "hip") or torch.version.hip is None:
        raise RuntimeError(f"HIP is not available; {label} benchmark requires ROCm.")


def run_conv2d_benchmark(
    batch_size,
    in_channels,
    out_channels,
    kernel_h,
    kernel_w,
    in_h,
    in_w,
    warmup,
    repeat,
    iters,
    validate,
):
    """
    Run conv2d benchmark.

    Args:
        batch_size: Batch size of input tensor
        in_channels: Number of input channels
        out_channels: Number of output channels
        kernel_h: Height of convolution kernel
        kernel_w: Width of convolution kernel
        in_h: Height of input image
        in_w: Width of input image
        warmup: Number of warmup iterations
        repeat: Number of benchmark repeats
        iters: Number of graph replays per timing repeat
        validate: Whether to validate output correctness
    """
    _ensure_rocm_available("conv2d_round2")

    stride_h = stride_w = kernel_module.STRIDE
    dilation = kernel_module.DILATION
    padding = kernel_module.PADDING
    groups = kernel_module.GROUPS

    supported = (
        batch_size == kernel_module.INPUT0_SHAPE[0]
        and in_channels == kernel_module.INPUT0_SHAPE[1]
        and out_channels == kernel_module.WEIGHT_SHAPE[0]
        and kernel_h == kernel_module.WEIGHT_SHAPE[2]
        and kernel_w == kernel_module.WEIGHT_SHAPE[3]
        and in_h == kernel_module.INPUT0_SHAPE[2]
        and in_w == kernel_module.INPUT0_SHAPE[3]
    )
    if not supported:
        raise RuntimeError(
            "This round kernel only supports "
            f"batch={kernel_module.INPUT0_SHAPE[0]}, "
            f"in_c={kernel_module.INPUT0_SHAPE[1]}, "
            f"out_c={kernel_module.WEIGHT_SHAPE[0]}, "
            f"kernel=({kernel_module.WEIGHT_SHAPE[2]},{kernel_module.WEIGHT_SHAPE[3]}), "
            f"in=({kernel_module.INPUT0_SHAPE[2]},{kernel_module.INPUT0_SHAPE[3]})."
        )

    out_h = (in_h + 2 * padding - dilation * (kernel_h - 1) - 1) // stride_h + 1
    out_w = (in_w + 2 * padding - dilation * (kernel_w - 1) - 1) // stride_w + 1

    input = torch.randn((batch_size, in_channels, in_h, in_w), dtype=torch.float32, device="cuda")
    weight = torch.randn(kernel_module.WEIGHT_SHAPE, dtype=torch.float32, device="cuda")

    model = kernel_module.ModelNew(
        in_channels,
        out_channels,
        kernel_h,
        stride=stride_h,
        padding=padding,
        dilation=dilation,
        groups=groups,
        bias=False,
    ).cuda()
    with torch.no_grad():
        model.conv2d.weight.copy_(weight)
        model._get_cached_weight(input)

    out = torch.empty((batch_size, out_channels, out_h, out_w), dtype=kernel_module.OUTPUT_TORCH_DTYPE, device="cuda")

    for _ in range(warmup):
        out.copy_(model(input))
    torch.cuda.synchronize()

    graph = torch.cuda.CUDAGraph()
    capture_stream = torch.cuda.Stream()
    with torch.cuda.graph(graph, stream=capture_stream):
        out.copy_(model(input))
    torch.cuda.synchronize()

    start_evt = torch.cuda.Event(enable_timing=True)
    end_evt = torch.cuda.Event(enable_timing=True)
    start_evt.record()
    for _ in range(repeat):
        for _ in range(iters):
            graph.replay()
    end_evt.record()
    torch.cuda.synchronize()

    elapsed_ms = start_evt.elapsed_time(end_evt)
    elapsed = (elapsed_ms * 1.0e-3) / (repeat * iters)

    flops_per_output = kernel_module.WEIGHT_SHAPE[1] * kernel_h * kernel_w * 2  # mul + add
    total_flops = batch_size * out_channels * out_h * out_w * flops_per_output
    tflops = total_flops / elapsed / 1.0e12

    input_bytes = batch_size * in_channels * in_h * in_w * 4
    weight_bytes = (
        kernel_module.WEIGHT_SHAPE[0]
        * kernel_module.WEIGHT_SHAPE[1]
        * kernel_h
        * kernel_w
        * 4
    )
    output_bytes = (
        batch_size
        * out_channels
        * out_h
        * out_w
        * torch.empty((), dtype=kernel_module.OUTPUT_TORCH_DTYPE).element_size()
    )
    bytes_moved = input_bytes + weight_bytes + output_bytes
    bandwidth = bytes_moved / elapsed / 1.0e9

    print(
        f"batch={batch_size} in_c={in_channels} out_c={out_channels} "
        f"kernel=({kernel_h},{kernel_w}) in=({in_h},{in_w}) out=({out_h},{out_w}) "
        f"time_ms={elapsed * 1.0e3:.4f} tflops={tflops:.3f} bandwidth_gbs={bandwidth:.3f}"
    )

    if validate:
        expected = torch.nn.functional.conv2d(
            input,
            weight,
            stride=(stride_h, stride_w),
            padding=padding,
            dilation=dilation,
            groups=groups,

        max_abs = torch.max(torch.abs(actual - expected)).item()
        if not torch.allclose(actual, expected, rtol=2e-2, atol=2e-2):

def kernel_amdgpu_raw_buffer_roundtrip(

    dst: S.Tensor((7,), S.i32),
    range_bytes: S.i32,
):
    src_rsrc = S.amdgpu.make_rsrc(src, range_bytes)
    dst_rsrc = S.amdgpu.make_rsrc(dst, range_bytes)

    zero = S.convert(0, S.i32)
    offset_x2 = S.convert(4, S.i32)
    offset_x4 = S.convert(12, S.i32)

    value_x1 = S.amdgpu.raw_buffer_load_x1(src_rsrc, zero, zero, 0)
    value_x2 = S.amdgpu.raw_buffer_load_x2(src_rsrc, zero, offset_x2, 0)
    value_x4 = S.amdgpu.raw_buffer_load_x4(src_rsrc, zero, offset_x4, 0)

    S.amdgpu.raw_buffer_store_x1(value_x1, dst_rsrc, zero, zero, 0)
    S.amdgpu.raw_buffer_store_x2(value_x2, dst_rsrc, zero, offset_x2, 0)
    S.amdgpu.raw_buffer_store_x4(value_x4, dst_rsrc, zero, offset_x4, 0)

def generate_mlir(jit_fn) -> str:
    jit_deps = _collect_jit_dependencies(jit_fn)
    import_module = _build_import_module([jit_fn, *jit_deps])

    generator = _C.MLIRGenerator()
    generator.generate_from_python_ast(import_module)

    for dep in jit_deps:
        dep_func = _get_function_def(dep.parse())
        generator.visit_function_def(dep_func, "[]", "jit")

    kernel_func = _get_function_def(jit_fn.parse())
    generator.visit_function_def(kernel_func, "[]", "kernel")
    return generator.get_mlir()

    batch_size,
    in_channels,
    out_channels,
    kernel_h,
    kernel_w,
    in_h,
    in_w,
    warmup,
    repeat,
    iters,
    validate,

        in_channels: Number of input channels
        out_channels: Number of output channels
        kernel_h: Height of convolution kernel
        kernel_w: Width of convolution kernel
        in_h: Height of input image
        in_w: Width of input image
        warmup: Number of warmup iterations
        repeat: Number of benchmark repeats
        iters: Number of graph replays per timing repeat
        validate: Whether to validate output correctness

    out_w = (in_w + 2 * padding - dilation * (kernel_w - 1) - 1) // stride_w + 1
