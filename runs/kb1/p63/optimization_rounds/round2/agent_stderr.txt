def _load_round_module():
    spec = importlib.util.spec_from_file_location("kb_round1_output_model_new", ROUND_OUTPUT)
    if spec is None or spec.loader is None:
        raise RuntimeError(f"Unable to load round module from {ROUND_OUTPUT}.")
    module = importlib.util.module_from_spec(spec)
    spec.loader.exec_module(module)
    return module


def _generate_mlir(jit_fn) -> str:
    jit_deps = _collect_jit_dependencies(jit_fn)
    import_module = _build_import_module([jit_fn, *jit_deps])

    generator = _C.MLIRGenerator()
    generator.generate_from_python_ast(import_module)

    for dep in jit_deps:
        dep_func = _get_function_def(dep.parse())
        generator.visit_function_def(dep_func, "[]", "jit")

    kernel_func = _get_function_def(jit_fn.parse())
    generator.visit_function_def(kernel_func, "[]", "kernel")
    return generator.get_mlir()


@unittest.skipUnless(has_rocm(), "Requires ROCm/HIP with an AMD GPU.")
class TestConv2DAsym(unittest.TestCase):
    def test_weight_cache_reuses_device_tensor_until_storage_changes(self):
        module = _load_round_module()
        model = module.ModelNew(32, 64, (5, 9), stride=1, padding=0, dilation=1, groups=1, bias=False).cuda()
        x = torch.empty(module.INPUT0_SHAPE, dtype=torch.float32, device="cuda")

        with torch.no_grad():
            cached0 = model._get_cached_weight(x)
            cached1 = model._get_cached_weight(x)
            self.assertEqual(cached0.data_ptr(), cached1.data_ptr())

            model.conv2d.weight = torch.nn.Parameter(model.conv2d.weight.detach().clone())
            cached2 = model._get_cached_weight(x)
            self.assertNotEqual(cached1.data_ptr(), cached2.data_ptr())

    def test_kernel_mlir_contains_mfma(self):
        module = _load_round_module()
        mlir = _generate_mlir(module.fused_kernel)

def _ensure_rocm_available(label):
    """Check that ROCm/HIP is available."""
    if not torch.cuda.is_available():
        raise RuntimeError("CUDA is not available.")
    if not hasattr(torch.version, "hip") or torch.version.hip is None:
        raise RuntimeError(f"HIP is not available; {label} benchmark requires ROCm.")


def run_conv2d_benchmark(
    batch_size,
    in_channels,
    out_channels,
    kernel_h,
    kernel_w,
    in_h,
    in_w,
    warmup,
    repeat,
    iters,
    validate,
):
    """
    Run conv2d benchmark.

    Args:
        batch_size: Batch size of input tensor
        in_channels: Number of input channels
        out_channels: Number of output channels
        kernel_h: Height of convolution kernel
        kernel_w: Width of convolution kernel
        in_h: Height of input image
        in_w: Width of input image
        warmup: Number of warmup iterations
        repeat: Number of benchmark repeats
        iters: Number of graph replays per timing repeat
        validate: Whether to validate output correctness
    """
    _ensure_rocm_available("conv2d_round1")

    # Compute output dimensions
    stride_h = stride_w = 1
    dilation = 1
    padding = 0

    out_h = (in_h + 2 * padding - dilation * (kernel_h - 1) - 1) // stride_h + 1
    out_w = (in_w + 2 * padding - dilation * (kernel_w - 1) - 1) // stride_w + 1

    # Create input tensors
    input = torch.randn((batch_size, in_channels, in_h, in_w), dtype=torch.float32, device="cuda")
    weight = torch.randn((out_channels, in_channels, kernel_h, kernel_w), dtype=torch.float32, device="cuda")

    model = kernel_module.ModelNew(
        in_channels,
        out_channels,
        (kernel_h, kernel_w),
        stride=stride_h,
        padding=padding,
        dilation=dilation,
        groups=1,
        bias=False,
    ).cuda()
    with torch.no_grad():
        model.conv2d.weight.copy_(weight)
        model._get_cached_weight(input)

    out = torch.empty((batch_size, out_channels, out_h, out_w), dtype=torch.float32, device="cuda")

    for _ in range(warmup):
        out.copy_(model(input))
    torch.cuda.synchronize()

    graph = torch.cuda.CUDAGraph()
    capture_stream = torch.cuda.Stream()
    with torch.cuda.graph(graph, stream=capture_stream):
        out.copy_(model(input))
    torch.cuda.synchronize()

    start_evt = torch.cuda.Event(enable_timing=True)
    end_evt = torch.cuda.Event(enable_timing=True)
    start_evt.record()
    for _ in range(repeat):
        for _ in range(iters):
            graph.replay()
    end_evt.record()
    torch.cuda.synchronize()

    elapsed_ms = start_evt.elapsed_time(end_evt)
    elapsed = (elapsed_ms * 1.0e-3) / (repeat * iters)

    flops_per_output = in_channels * kernel_h * kernel_w * 2  # mul + add
    total_flops = batch_size * out_channels * out_h * out_w * flops_per_output
    tflops = total_flops / elapsed / 1.0e12

    input_bytes = batch_size * in_channels * in_h * in_w * 4
    weight_bytes = out_channels * in_channels * kernel_h * kernel_w * 4
    output_bytes = batch_size * out_channels * out_h * out_w * 4
    bytes_moved = input_bytes + weight_bytes + output_bytes
    bandwidth = bytes_moved / elapsed / 1.0e9

    print(
        f"batch={batch_size} in_c={in_channels} out_c={out_channels} "
        f"kernel=({kernel_h},{kernel_w}) in=({in_h},{in_w}) out=({out_h},{out_w}) "
        f"time_ms={elapsed * 1.0e3:.4f} tflops={tflops:.3f} bandwidth_gbs={bandwidth:.3f}"
    )

    if validate:
        expected = torch.nn.functional.conv2d(input, weight, stride=(stride_h, stride_w), padding=padding, dilation=dilation)
        actual = out
        max_abs = torch.max(torch.abs(actual - expected)).item()
        if not torch.allclose(actual, expected, rtol=1e-4, atol=1e-4):
            raise AssertionError(f"Validation failed (max abs diff {max_abs}).")
        print(f"validation=max_abs_diff:{max_abs:.6f}")

def generate_mlir(jit_fn) -> str:
    jit_deps = _collect_jit_dependencies(jit_fn)
    import_module = _build_import_module([jit_fn, *jit_deps])

    generator = _C.MLIRGenerator()
    generator.generate_from_python_ast(import_module)

    for dep in jit_deps:
        dep_func = _get_function_def(dep.parse())
        generator.visit_function_def(dep_func, "[]", "jit")

    kernel_func = _get_function_def(jit_fn.parse())
    generator.visit_function_def(kernel_func, "[]", "kernel")
    return generator.get_mlir()
