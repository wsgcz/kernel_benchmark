import torch
import torch.nn as nn
import substrate
import substrate.language as S

SQRT_2 = 1.4142135623730951

BATCH_SIZE = 1024
IN_FEATURES = 8192
OUT_FEATURES = 8192
SUBTRACT_VALUE = 2.0
MULTIPLY_VALUE = 1.5

BLOCK_M = 64
BLOCK_N = 64
WAVE_SIZE = 64
WAVES_PER_BLOCK = 4
K_TILE = 16
BF16_BYTES = 2


def _launch():
    return ((OUT_FEATURES // BLOCK_N, BATCH_SIZE // BLOCK_M, 1), (WAVE_SIZE * WAVES_PER_BLOCK, 1, 1))


@substrate.jit
def fused_kernel(
    X: S.Tensor((BATCH_SIZE, IN_FEATURES), S.bf16),
    W: S.Tensor((IN_FEATURES, OUT_FEATURES), S.bf16),
    BIAS: S.Tensor((OUT_FEATURES,), S.bf16),
    Y: S.Tensor((BATCH_SIZE, OUT_FEATURES), S.bf16),
):
    tid = S.thread_id(0)
    lane = tid % WAVE_SIZE
    wave = tid // WAVE_SIZE
    warp_row = wave // 2
    warp_col = wave % 2

    block_row = S.block_id(1) * BLOCK_M
    block_col = S.block_id(0) * BLOCK_N
    tile_row = block_row + warp_row * 32
    tile_col = block_col + warp_col * 32

    zero = S.convert(0, S.i32)
    x_rsrc = S.amdgpu.make_rsrc(X, BATCH_SIZE * IN_FEATURES * BF16_BYTES)
    w_rsrc = S.amdgpu.make_rsrc(W, IN_FEATURES * OUT_FEATURES * BF16_BYTES)

    a_shared = S.make_shared((WAVES_PER_BLOCK, WAVE_SIZE, 4), S.u32)
    b_shared = S.make_shared((WAVES_PER_BLOCK, WAVE_SIZE, 4), S.u32)

    acc = S.full((16,), 0.0, S.f32)

    a_row = lane % 32
    a_half = lane // 32
    a_elem = (tile_row + a_row) * IN_FEATURES
    a_word_base = a_half * 2

    b_k = lane // 4
    b_chunk = lane % 4
    b_g0 = b_chunk * 2
    b_g1 = b_g0 + 1
    b_lane0 = b_g0 * 8 + (b_k % 8)
    b_lane1 = b_g1 * 8 + (b_k % 8)
    b_word_base = (b_k // 8) * 2

    for k_base in S.range(IN_FEATURES // K_TILE):
        k_off = k_base * K_TILE

        a_byte_offset = (a_elem + k_off + a_half * 8) * BF16_BYTES
        a_pack = S.amdgpu.raw_buffer_load_x4(x_rsrc, zero, a_byte_offset, 0)

        a_shared[wave, a_row, a_word_base + 0] = a_pack[0]
        a_shared[wave, a_row, a_word_base + 1] = a_pack[1]
        a_shared[wave, a_row + 32, a_word_base + 0] = a_pack[2]
        a_shared[wave, a_row + 32, a_word_base + 1] = a_pack[3]

        b_byte_offset = ((k_off + b_k) * OUT_FEATURES + tile_col + b_chunk * 8) * BF16_BYTES
        b_pack = S.amdgpu.raw_buffer_load_x4(w_rsrc, zero, b_byte_offset, 0)

        b_shared[wave, b_lane0, b_word_base + 0] = b_pack[0]
        b_shared[wave, b_lane0, b_word_base + 1] = b_pack[1]
        b_shared[wave, b_lane1, b_word_base + 0] = b_pack[2]
        b_shared[wave, b_lane1, b_word_base + 1] = b_pack[3]

        S.syncthreads()

        a_frag = S.view(a_shared[wave, lane], S.Tensor((2, 4, 1), S.bf16))
        b_frag = S.view(b_shared[wave, lane], S.Tensor((2, 4, 1), S.bf16))

        acc = S.amdgpu.mfma_32x32x8_bf16_f32(a_frag[0], b_frag[0], acc)
        acc = S.amdgpu.mfma_32x32x8_bf16_f32(a_frag[1], b_frag[1], acc)

        S.syncthreads()

    lane_col = lane % 32
    lane_row_group = lane // 32
    bias_val = S.convert(BIAS[tile_col + lane_col], S.f32)
    sub_val = S.convert(SUBTRACT_VALUE, S.f32)
    mul_val = S.convert(MULTIPLY_VALUE, S.f32)
    zero_f32 = S.convert(0.0, S.f32)

    for acc_idx in S.range(16):
        row = tile_row + 8 * (acc_idx // 4) + 4 * lane_row_group + (acc_idx % 4)
        col = tile_col + lane_col
        out = (acc[acc_idx] + bias_val - sub_val) * mul_val
        if out > zero_f32:
            Y[row, col] = S.convert(out, S.bf16)
        else:
            Y[row, col] = S.convert(0.0, S.bf16)
