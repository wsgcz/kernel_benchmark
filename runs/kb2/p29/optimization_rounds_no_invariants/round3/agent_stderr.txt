#!/usr/bin/env python3
import unittest

import torch

import substrate
import substrate.language as S
from substrate.testing import has_rocm


BF16_BYTES = 2


@substrate.jit
def kernel_store_x4_full_rsrc(
    src: S.Tensor((1, 4), S.u32),
    out_ptr: S.Pointer(S.bf16),
    m: S.u32,
    n: S.u32,
):
    out_memref = S.make_tensor(out_ptr, S.bf16, S.make_layout((m * n,), (1,)))
    out_rsrc = S.amdgpu.make_rsrc(out_memref, m * n * BF16_BYTES)
    zero = S.convert(0, S.u32)
    S.amdgpu.raw_buffer_store_x4(src[0], out_rsrc, zero, zero, 0)


@substrate.jit
def kernel_store_x4_row_subview_rsrc(
    src: S.Tensor((1, 4), S.u32),
    out_ptr: S.Pointer(S.bf16),
    m: S.u32,
    n: S.u32,
    row: S.u32,
):
    out_memref = S.make_tensor(out_ptr, S.bf16, S.make_layout((m * n,), (1,)))
    row_view = S.subview(out_memref, (row * n,), (n,), (1,))
    row_rsrc = S.amdgpu.make_rsrc(row_view, n * BF16_BYTES)
    zero = S.convert(0, S.u32)
    S.amdgpu.raw_buffer_store_x4(src[0], row_rsrc, zero, zero, 0)

    def test_row_subview_rsrc_honors_subview_offset(self):

        out_subview = torch.zeros((4, 8), dtype=torch.bfloat16, device="cuda")

NUMEL = 4096
THREADS = 256
BLOCKS = NUMEL // THREADS
L2E = 1.442695041
SMALL_THRESHOLD = 4.997253418e-3


@substrate.jit
def fast_tanhf(a: S.f32) -> S.f32:
    zero = S.convert(0.0, S.f32)
    one = S.convert(1.0, S.f32)
    neg_two_l2e = S.convert(-2.0 * L2E, S.f32)
    small = S.convert(SMALL_THRESHOLD, S.f32)

    s = S.abs(a)

    e = S.exp2(neg_two_l2e * s)
    r = S.amdgpu.rcp(e + one)
    r = r - e * r

    r = -r if a < zero else r
    return a if s < small else r


@substrate.jit
def kernel_fast_tanh_approx(
    inp: S.Tensor((NUMEL,), S.f32),
    out: S.Tensor((NUMEL,), S.f32),
):
    idx = S.block_id(0) * S.block_dim(0) + S.thread_id(0)
    out[idx] = fast_tanhf(inp[idx])
