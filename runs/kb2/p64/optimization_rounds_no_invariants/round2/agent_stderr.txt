NEGATIVE_SLOPE = 0.01

_DUMMY_A_RANGE_BYTES = 2 * 64 * 4 * 4
_DUMMY_B_RANGE_BYTES = 2 * 64 * 4 * 4
_DUMMY_C_RANGE_BYTES = 2 * 2 * 64 * 16 * 4


def _launch_mfma_touch():
    return ((1, 1, 1), (256, 1, 1))


@substrate.jit
def mfma_touch_kernel(
    a_global: S.Tensor((2, 64, 4), S.u32),
    b_global: S.Tensor((2, 64, 4), S.u32),
    c_global: S.Tensor((2, 2, 64, 16), S.f32),
):
    tid = S.thread_id(0)
    lane = tid % 64
    warp = tid // 64
    warp_row = warp // 2
    warp_col = warp % 2

    a_shared = S.make_shared((2, 64, 4), S.u32)
    b_shared = S.make_shared((2, 64, 4), S.u32)

    a_rsrc = S.amdgpu.make_rsrc(a_global, S.convert(_DUMMY_A_RANGE_BYTES, S.i32))
    b_rsrc = S.amdgpu.make_rsrc(b_global, S.convert(_DUMMY_B_RANGE_BYTES, S.i32))

    zero = S.convert(0, S.i32)
    a_offset = S.convert((warp_row * 64 + lane) * 16, S.i32)
    b_offset = S.convert((warp_col * 64 + lane) * 16, S.i32)

    a_words = S.amdgpu.raw_buffer_load_x4(a_rsrc, zero, a_offset, 0)
    b_words = S.amdgpu.raw_buffer_load_x4(b_rsrc, zero, b_offset, 0)

    for i in S.range(4):
        a_shared[warp_row, lane, i] = a_words[i]
        b_shared[warp_col, lane, i] = b_words[i]

    S.syncthreads()

    a_frag = S.view(a_shared[warp_row, lane], S.Tensor((2, 4, 1), S.bf16))
    b_frag = S.view(b_shared[warp_col, lane], S.Tensor((2, 4, 1), S.bf16))

    c_lane = S.full((16,), 0.0, S.f32)
    c_lane = S.amdgpu.mfma_32x32x8_bf16_f32(a_frag[0], b_frag[0], c_lane)
    c_lane = S.amdgpu.mfma_32x32x8_bf16_f32(a_frag[1], b_frag[1], c_lane)

    for i in S.range(16):
        c_global[warp_row, warp_col, lane, i] = c_lane[i]

REFERENCE_CASES = (
    ("h8_kv8_seq256", [flash_attn.BLOCK_ROWS], 8, 8),
    ("h8_kv8_seq512", [flash_attn.BLOCK_ROWS * 2], 8, 8),
    ("h8_kv1_seq256", [flash_attn.BLOCK_ROWS], 8, 1),
    ("h8_kv1_seq512", [flash_attn.BLOCK_ROWS * 2], 8, 1),
    ("h8_kv1_packed_seq256_512", [flash_attn.BLOCK_ROWS, flash_attn.BLOCK_ROWS * 2], 8, 1),

def flash_attn_reference_packed(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    seq_ptr: torch.Tensor,
) -> torch.Tensor:
    seq_ptr_host = seq_ptr.to(device="cpu", dtype=torch.int64)
    heads_q = q.shape[1]
    heads_kv = k.shape[1]
    gqa_ratio = heads_q // heads_kv
    out = torch.empty_like(q, dtype=torch.float32)

    for seq_id in range(seq_ptr_host.numel() - 1):
        seq_begin = int(seq_ptr_host[seq_id].item())
        seq_end = int(seq_ptr_host[seq_id + 1].item())
        q_seq = q[seq_begin:seq_end].to(torch.float32)
        k_seq = k[seq_begin:seq_end].to(torch.float32)
        v_seq = v[seq_begin:seq_end].to(torch.float32)
        seq_len = seq_end - seq_begin
        causal_mask = torch.triu(
            torch.ones((seq_len, seq_len), dtype=torch.bool, device=q.device),
            diagonal=1,
        )

        for head_idx in range(heads_q):
            kv_head_idx = head_idx // gqa_ratio
            q_head = q_seq[:, head_idx]
            k_head = k_seq[:, kv_head_idx]
            v_head = v_seq[:, kv_head_idx]

            qk = q_head @ k_head.transpose(0, 1)
            qk = qk.masked_fill(causal_mask, flash_attn.NEG_INF)
            row_max = qk.max(dim=1, keepdim=True).values
            row_max = torch.where(
                row_max == flash_attn.NEG_INF,
                torch.zeros_like(row_max),
                row_max,
            )
            ptilde = torch.exp2(qk * flash_attn.SCALE_LOG2 - row_max * flash_attn.SCALE_LOG2)
            row_sum = ptilde.sum(dim=1, keepdim=True)
            out[seq_begin:seq_end, head_idx] = (ptilde / row_sum) @ v_head

    return out.contiguous().to(torch.bfloat16).to(torch.float32)

class TestFlashAttention(unittest.TestCase):
    RTOL = 1e-2
    ATOL = 1e-2

    def setUp(self):
        torch.manual_seed(0)

    def _require_hip(self):
        if not torch.cuda.is_available():
            self.skipTest("CUDA is not available.")
        if not hasattr(torch.version, "hip") or torch.version.hip is None:
            self.skipTest("HIP is not available.")

    def _make_inputs(
        self,
        seq_lens: list[int],
        q_heads: int,
        kv_heads: int,
    ) -> tuple[torch.Tensor, torch.Tensor, torch.Tensor, torch.Tensor]:
        total_tokens = sum(seq_lens)
        q = torch.randn((total_tokens, q_heads, flash_attn.HEAD_DIM), dtype=torch.bfloat16, device="cuda")
        k = torch.randn((total_tokens, kv_heads, flash_attn.HEAD_DIM), dtype=torch.bfloat16, device="cuda")
        v = torch.randn((total_tokens, kv_heads, flash_attn.HEAD_DIM), dtype=torch.bfloat16, device="cuda")
        seq_ptr = torch.tensor(
            [0, *torch.tensor(seq_lens).cumsum(0).tolist()],
            dtype=torch.int32,
            device="cuda",
        )
        return q, k, v, seq_ptr

    def _assert_flash_attn_matches_reference(
        self,
        seq_lens: list[int],
        q_heads: int,
        kv_heads: int,
    ) -> None:
        self._require_hip()
        q, k, v, seq_ptr = self._make_inputs(seq_lens, q_heads, kv_heads)

        actual = flash_attn.flash_attn(q, k, v, seq_ptr).to(torch.float32).cpu()
        expected = flash_attn_reference_packed(q, k, v, seq_ptr).cpu()

        torch.testing.assert_close(
            actual,
            expected,
            rtol=self.RTOL,
            atol=self.ATOL,
            msg=f"Flash attention output does not match the host reference.\nExpected:\n{expected}\nActual:\n{actual}\n"
            f"Max absolute difference: {torch.max(torch.abs(actual - expected))}",
        )

    def test_flash_attn_bf16_reference_cases(self):
        # Match the BF16 cases this kernel currently supports in petit-kernel.
        for name, seq_lens, q_heads, kv_heads in REFERENCE_CASES:
            with self.subTest(name=name):
                self._assert_flash_attn_matches_reference(seq_lens, q_heads, kv_heads)

    def test_flash_attn_rejects_seq_len_not_multiple_of_block_rows(self):
        self._require_hip()

        v = torch.randn((seq_len, 1, flash_attn.HEAD_DIM), dtype=torch.bfloat16, device="cuda")
        seq_ptr = torch.tensor([0, seq_len], dtype=torch.int32, device="cuda")

        with self.assertRaisesRegex(NotImplementedError, "multiple of BLOCK_ROWS"):
            flash_attn.flash_attn(q, k, v, seq_ptr)

    MMA_N = 8
    MMA_K = 16

    WARPS_N = 2  # Number of warps in N dimension per work group

    MMA_TILES_N = 2  # Number of MMA tiles per warp in N dimension

    BLOCK_N = WARPS_N * MMA_TILES_N * MMA_N  # Total N covered by work group

    warp_col_id = warp_id % WARPS_N
